```typescript
interface User {
  id: number;
  name: string;
  email: string;
  age: number;
  role: "admin" | "user";
}
```

> **Partial** - makes all properties optional

```typescript
type UserUpdate = Partial<User>;
// { id?: number; name?: string; email?: string; age?: number; role?: "admin" | "user" }
```

> **Required** - makes all properties required

```typescript
type UserRequired = Required<User>;
// { id: number; name: string; email: string; age: number; role: "admin" | "user" }
```

> **Pick** - picks a subset of properties

```typescript
type UserPreview = Pick<User, "id" | "name">;
// { id: number; name: string }
```

> **Omit** - omits a subset of properties

```typescript
type UserWithoutEmail = Omit<User, "email">;
// { id: number; name: string; age: number; role: "admin" | "user" }
```

> **Readonly** - makes all properties readonly

```typescript
type ReadonlyUser = Readonly<User>;
// { readonly id: number; readonly name: string; readonly email: string; readonly age: number; readonly role: "admin" | "user" }
```

> **Record** - creates a type with specified keys and values

```typescript
type UserRoles = Record<"admin" | "user", string>;
// { admin: string; user: string }
type UserMap = Record<number, User>;
// { [key: number]: User }
```

> **ReturnType** and **Parameters** - get the return type / parameter types of a function

```typescript
// ReturnType - gets the return type of a function
function getUser() {
  return { id: 1, name: "Alice", email: "alice@example.com" };
}
type GetUserReturn = ReturnType<typeof getUser>;
// { id: number; name: string; email: string }

// Parameters - gets the parameter types of a function
function createUser(name: string, email: string) {
  return { id: Date.now(), name, email };
}
type CreateUserParams = Parameters<typeof createUser>;
// [string, string]
```

## Custom utility types

> Making specific keys optional, but make rest required

```typescript
type PartialBy<T, K extends keyof T> = Omit<T, K> & Partial<Pick<T, K>>;

interface CreateIssueInput {
  title: string;
  description: string;
  assigneeId: number;
  priority: "low" | "medium" | "high";
}

type CreateIssueForm = PartialBy<CreateIssueInput, "assigneeId" | "priority">;
// { title: string; description: string; assigneeId?: number; priority?: "low" | "medium" | "high" }
```
